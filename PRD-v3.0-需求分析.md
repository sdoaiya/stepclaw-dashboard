## 1. 项目概述

### 1.1 版本演进

```
v1.0 (基础版) → v2.0 (静态架构) → v2.1 (视觉优化) → v3.0 (智能看板)
     │               │                │                │
     ▼               ▼                ▼                ▼
  基础展示       纯前端架构      玻璃态三栏布局    全生命周期管理
  静态数据       GitHub Pages    部门筛选/动效     +学习优化+监控+日报
```

### 1.2 v3.0 核心目标

在 v2.1 视觉优化的基础上，引入**四大核心能力**：

| 核心模块 | 英文代号 | 核心价值 | 目标用户 |
|---------|---------|---------|---------|
| **任务全生命周期管理** | TLC (Task Lifecycle) | 任务从创建到归档的完整追踪 | 项目经理/团队Leader |
| **学习优化系统** | LOS (Learning Optimization) | 基于历史数据的智能建议 | 全体成员 |
| **监控自检系统** | MSS (Monitoring & Self-check) | 系统健康度实时监控 | 运维/管理员 |
| **日报展示系统** | DRS (Daily Report System) | 自动化日报生成与展示 | 管理层/全员 |

### 2.2 任务数据结构

```typescript
interface Task {
  // 基础信息
  id: string;                    // 任务唯一标识 (task_xxx)
  title: string;                 // 任务标题
  description: string;           // 任务描述 (Markdown支持)
  
  // 生命周期状态
  status: TaskStatus;            // 当前状态
  statusHistory: StatusChange[]; // 状态变更历史
  
  // 人员分配
  creator: EmployeeRef;          // 创建者
  assignee?: EmployeeRef;        // 负责人
  collaborators: EmployeeRef[];  // 协作者
  reviewer?: EmployeeRef;        // 验收人
  
  // 时间管理
  createdAt: string;             // 创建时间 (ISO 8601)
  startedAt?: string;            // 开始处理时间
  completedAt?: string;          // 完成时间
  dueDate?: string;              // 截止日期
  estimatedHours?: number;       // 预估工时
  actualHours?: number;          // 实际工时
  
  // 优先级与分类
  priority: 'urgent' | 'high' | 'medium' | 'low';
  category: TaskCategory;        // 任务分类
  tags: string[];                // 标签
  
  // 部门与项目
  dept: Department;              // 所属部门
  project?: string;              // 所属项目
  
  // 关联关系
  parentTask?: string;           // 父任务ID
  subtasks: string[];            // 子任务ID列表
  dependencies: string[];        // 依赖任务ID列表
  relatedDocs: string[];         // 关联文档链接
  
  // 进度追踪
  progress: number;              // 进度百分比 (0-100)
  checklists: ChecklistItem[];   // 检查清单
  
  // 质量指标
  quality?: QualityMetrics;      // 质量评估
  reviewComments?: Comment[];    // 验收意见
  
  // 学习优化数据
  similarTasks?: string[];       // 相似任务ID
  suggestedAssignee?: string;    // AI建议负责人
  riskLevel?: 'low' | 'medium' | 'high';  // 风险等级
}

interface StatusChange {
  from: TaskStatus;
  to: TaskStatus;
  timestamp: string;
  operator: EmployeeRef;
  reason?: string;               // 变更原因
}

interface ChecklistItem {
  id: string;
  content: string;
  completed: boolean;
  completedAt?: string;
}

interface QualityMetrics {
  codeQuality?: number;          // 代码质量评分 (0-100)
  docQuality?: number;           // 文档质量评分
  onTimeDelivery: boolean;       // 是否按时交付
  revisionCount: number;         // 返工次数
}
```

### 3.2 核心模型

#### 3.2.1 工时预估模型 (ETM - Estimated Time Model)

```typescript
interface TimeEstimation {
  taskId: string;
  estimatedHours: number;           // 预估工时
  confidence: number;               // 置信度 (0-1)
  
  // 影响因素
  factors: {
    historicalSimilarity: number;   // 历史相似任务平均工时
    complexity: number;             // 任务复杂度评分
    assigneeSkill: number;          // 负责人技能匹配度
    dependencies: number;           // 依赖任务影响
    riskBuffer: number;             // 风险缓冲时间
  };
  
  // 区间预估
  range: {
    optimistic: number;             // 乐观估计 (P10)
    pessimistic: number;            // 悲观估计 (P90)
    mostLikely: number;             // 最可能 (P50)
  };
  
  // 相似任务参考
  similarTasks: {
    taskId: string;
    similarity: number;             // 相似度 (0-1)
    actualHours: number;
    variance: number;               // 偏差
  }[];
}
```

**预估算法逻辑**:

```
预估工时 = 基础工时 × 复杂度系数 × 人员系数 × 依赖系数 + 风险缓冲

其中:
- 基础工时 = 历史相似任务平均实际工时
- 复杂度系数 = 1 + (任务复杂度评分 - 5) × 0.1
- 人员系数 = 1 / (负责人相关技能熟练度)
- 依赖系数 = 1 + (依赖任务数量 × 0.05)
- 风险缓冲 = 预估工时 × 风险等级系数
```

#### 3.2.2 人员匹配模型 (PMM - People Matching Model)

```typescript
interface PeopleMatch {
  taskId: string;
  candidates: {
    employee: EmployeeRef;
    matchScore: number;             // 匹配分数 (0-100)
    
    // 匹配维度
    dimensions: {
      skillMatch: number;           // 技能匹配度
      availability: number;         // 可用性 (当前负载)
      historicalPerformance: number; // 历史表现
      interest: number;             // 兴趣匹配
      collaboration: number;        // 协作历史
    };
    
    // 推荐理由
    reasons: string[];
    
    // 风险提示
    risks: {
      type: string;
      level: 'low' | 'medium' | 'high';
      description: string;
    }[];
  }[];
}
```

**匹配维度权重**:

interface DailyReport {

  id: string;                    // 报告ID

  // 报告主体
  summary: ReportSummary;        // 摘要
  sections: ReportSection[];     // 详细章节
  
  // 统计数据
  stats: ReportStats;            // 统计数据
  
  // 元数据
  recipients: string[];          // 接收人
  isAutoGenerated: boolean;      // 是否自动生成
  aiAssisted: boolean;           // 是否AI辅助生成
}

interface ReportSummary {
  title: string;                 // 报告标题
  highlights: string[];          // 今日亮点
  concerns: string[];            // 关注事项
  tomorrow: string[];            // 明日计划
}

interface ReportSection {
  title: string;                 // 章节标题
  icon: string;                  // 图标
  content: string;               // 内容 (Markdown)
  data?: any;                    // 关联数据
}

interface ReportStats {
  // 任务统计
  tasks: {
    created: number;
    completed: number;

    commits: number;

    linesChanged: number;
  };
  
  // 文档统计
  docs?: {

interface DailyReportConfig {
  // 生成时间
  schedule: {
    personal: string;            // 个人日报生成时间 "18:00"
    team: string;                // 团队日报生成时间 "19:00"
    weekly: string;              // 周报生成时间 "周五 18:00"
    monthly: string;             // 月报生成时间 "月末 18:00"
  };
  
  // 内容配置
  content: {
    includeCodeStats: boolean;   // 包含代码统计
    includeTaskDetails: boolean; // 包含任务详情
    includeAiSummary: boolean;   // 包含AI生成摘要
    maxTasksShown: number;       // 最大展示任务数
    language: 'zh' | 'en';       // 报告语言
  };
  
  // 分发配置
  distribution: {
    personal: {
      autoSend: boolean;
      channels: ('im' | 'email')[];
      recipients: string[];      // 默认发送给本人和直属上级
    };
    team: {
      autoSend: boolean;
      channels: ('im' | 'email')[];
      recipients: string[];      // 团队群/部门群
    };
  };
  
  // AI 辅助配置
  ai: {
    enabled: boolean;
    provider: 'openai' | 'local';
    model?: string;
    temperature: number;
    maxTokens: number;
  };
}

    },
    team: {
